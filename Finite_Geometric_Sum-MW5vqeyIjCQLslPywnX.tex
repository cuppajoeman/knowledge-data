\begin{prop*}{Finite Geometric Sum}
Finite Geometric sum, for any $x \ge 0$:
\[
\sum_{i=0}^{k} x^{i}=\frac{x^{k + 1} - 1}{x - 1}
\]

\begin{pf}
Let $S = \sum_{i=0}^k x^i$, then we have 
\[
xS - S =  \sum_{i=0}^k x^{i+1} - \sum_{i=0}^k x^i = x^{k+1} - x^0 = x^{k+1} - 1
\]
therefore 
\[
S = \frac{x^{k+1} - 1}{x-1}
\]
\end{pf}

\end{prop*}
