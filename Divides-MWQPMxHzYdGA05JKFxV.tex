\begin{defn*}{Divides}
  For $a, b \in \mathbb{Z}$ we say that $a$ divides $b$ and write $a \mid b$ if and only if there exists $k \in \mathbb{Z}$ such that 
  \[
  b = k  \cdot a
  \]
\end{defn*}
