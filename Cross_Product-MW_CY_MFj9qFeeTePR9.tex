\begin{defn*}{Cross Product}
  The cross product of two vectors $ \vec{a}, \vec{b} \in \mathbb{R} ^{3}$ is a vector $ \vec{c}$ perpendicular to both $ \vec{a}$ and $ \vec{b}$ and whose magnitude is equal to the area of the parallelogram generated by $ \vec{a}$ and $ \vec{b}$: 
  \[
  \vec{a} \times \vec{b} \stackrel{\mathtt{D}}{=} \left\Vert a \right\Vert \left\Vert b \right\Vert \sin  \left( \theta \right) \vec{n}
  \]
  with $\theta$ being the angle in the range $ \left[ 0, \pi \right]$ between $ \vec{a}$ and $ \vec{b}$ and $ \vec{n}$ a unit vector that is normal to both $ \vec{a}$ and $ \vec{b}$  oriented with respect to the right hand rule.
\end{defn*}
