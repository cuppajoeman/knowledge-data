\begin{defn*}{Interior of a Set}
  For $S \subseteq \mathbb{R} ^{n}$ we have the interior of $S$: 
  \[
    \mathring{S} \stackrel{\mathtt{D}}{=} \left\{ x \in \mathbb{R} ^{n}: \exists \varepsilon \in \mathbb{R} ^{>0}, B\left(x, \varepsilon\right) \subseteq S  \right\} 
  \]
\end{defn*}
