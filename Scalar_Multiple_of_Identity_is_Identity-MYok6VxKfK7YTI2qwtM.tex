\begin{theo*}{Scalar Multiple of Identity is Identity}
  \[
  a0 = 0 \text{ for every $ a \in \mathbb{F}$  } 
  \]
  \begin{pf}
    \[
     a 0 =  a \left( 0  +  0 \right) =  a0  +  a0
    \]
    So by adding the inverse of $ a0$ to each side, you get $ 0 =  a0$, as needed.
  \end{pf}
\end{theo*}
