\begin{defn*}{Constant Function in R}
  A constant function is a function $f$ such that there exists an element $C \in \mathbb{R}$ such that $\forall x \in \operatorname{dom}\left(f\right)$ 
  \[
  f\left(x\right) = C
  \]
\end{defn*}
