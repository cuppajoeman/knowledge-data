\begin{defn*}{Closure of a Set}
  Given $S \subseteq \mathbb{R} ^{n}$ the closure of $S$  is 
  \[
 \overline{S} \stackrel{\mathtt{D}}{=} \left\{ x \in \mathbb{R} ^{n}: \forall \varepsilon \in \mathbb{R} ^{>0}, B\left(x, \varepsilon\right) \cap S \neq \varnothing \right\} 
  \]
\end{defn*}
