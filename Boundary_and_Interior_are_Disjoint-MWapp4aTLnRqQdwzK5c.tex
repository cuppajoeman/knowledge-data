\begin{prop*}{Boundary and Interior are Disjoint}
  \[
  \partial S \cap \mathring{S} = \varnothing
  \]
  \begin{pf}
    Let $x \in \partial S \stackrel{\mathtt{D}}{=} \overline{S} \setminus \mathring{S}$, therefore $x \not\in \mathring{S}$. Thus it is impossible to have an element both in $ \partial S$ and $\mathring{S}$ at once.
  \end{pf}
\end{prop*}
