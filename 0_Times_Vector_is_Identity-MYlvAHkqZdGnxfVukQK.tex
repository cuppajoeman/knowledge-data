\begin{theo*}{0 Times Vector is Identity}
  \[
  0 v =  0 \text{ for every $  v \in V$  } 
  \]
  \begin{pf}
   \[
   0v = \left( 0  +  0  \right)v =  0v  + 0v
   \]
   by distributivity, then adding the inverse of $ 0v$ to both sides tells us that
   \[
   0 =  0v
   \]
  \end{pf}
\end{theo*}
