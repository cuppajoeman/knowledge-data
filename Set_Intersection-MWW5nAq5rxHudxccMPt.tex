\begin{defn*}{Set Intersection}
  For two sets $A, B$ their union is the set of all elements that they share, that is
  \[
  A \cap B \stackrel{\mathtt{D}}{=} \left\{ x : x \in A \text{ and } x \in B \right\}
  \]
\end{defn*}
