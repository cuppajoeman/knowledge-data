\begin{def*}{Boundary of a Set}
  Given $S \subseteq \mathbb{R} ^{n}$, the boundary of $S$ is 
  \[
  \partial S \stackrel{\mathtt{D}}{=} \overline{S} \setminus \mathring{S}
  \]
\end{def*}
