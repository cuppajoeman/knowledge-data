\begin{defn*}{Set Complement}
Given a set $U$ and $X \subseteq U$, the complement of $X$ is:
\[
	X ^{\complement} \stackrel{\text{def}}{=} \left\{ x \in U: x \not\in X \right\}
\]
\end{defn*}
