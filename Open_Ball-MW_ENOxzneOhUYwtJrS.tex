\begin{defn*}{Open Ball}
Let $\left( M, d \right)$ be a metric space, an open ball of radius $\varepsilon > 0$: 
\[
	B\left(x,\varepsilon\right)\stackrel{\text{def}}{=} \left\{ y: d\left(x,y\right)< \varepsilon \right\}
\]
\end{defn*}
