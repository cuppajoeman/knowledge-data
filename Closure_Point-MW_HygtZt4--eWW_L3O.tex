\begin{defn*}{Closure Point}
  For $S \subseteq \mathbb{R} ^{n}$ we say that $x \in \mathbb{R} ^{n }$ is a closure point of $S$ if 
  \[
  \forall \varepsilon \in \mathbb{R} ^{> 0}, B\left(x, \varepsilon\right) \cap S  \neq \varnothing
  \]
\end{defn*}
