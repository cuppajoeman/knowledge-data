\begin{defn*}{Congruence}
  \begin{itemize}
    \item For $a, b, \in \mathbb{Z}$ and $n \in \mathbb{N}$ we say that $a$ is congruent to $b$ modulo $n$ and write $a  \equiv b \mod n$ when 
      \[
      n \mid a - b
      \]
    \item A fundamental observation is that $a$ and $b$ are congruent if and only if they have the same remainder modulo $n$
  \end{itemize}

\end{defn*}
