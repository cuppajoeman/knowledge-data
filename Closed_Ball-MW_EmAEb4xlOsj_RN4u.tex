\begin{defn*}{Closed Ball}
Let $\left( M, d \right)$ be a metric space, a closed ball of radius $\varepsilon > 0$:
\[
	\overline{B} \left(x,y\right)\stackrel{\mathtt{D}}{=} \left\{ y: d\left(x,y\right) \le \varepsilon \right\}
\]
\end{defn*}
