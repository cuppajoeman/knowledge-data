\begin{defn*}{Bounded Subset of $ \mathbb{R} ^{n}$ }
  A subset $S \subseteq \mathbb{R} ^{n}$ is bounded if there exists $r \in \mathbb{R} ^{> 0}$ such that $S \subseteq B\left( \vec{0}, r\right)$ 
\end{defn*}
