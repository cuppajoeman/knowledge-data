\begin{theo*}{minus 1 times a Vector is its Inverse}
  \[
  \left( -1 \right) v =   - v \text{ for every $ v \in V$  } 
  \]
  \begin{pf}
    For $  v \in V$, we have
    \[
    v  +  \left( -1 \right)v =  1v  +  \left( -1 \right)v =  \left( 1  +  -1 \right)v =  0v =  0
    \]
    Therefore $ \left( -1 \right)v$ is the additive inverse of $ v$. 
  \end{pf}
\end{theo*}
